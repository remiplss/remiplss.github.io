    import React from "react";
import { useEffect, useState } from "react";
import axios from "axios";
import { URLDayl } from "../../services/constantes"
import { Link } from "react-router-dom"
import ListModal from "../general/ListModal";

/**
 * This component contain your list of likes
 * @param {object} user - user object from api
 * @param {object} setUser - user setter
 * @emits - paintingEntity/{id} : get paint informations
 * @emits - users/updateFavoritePainting : delete paint
 */

export default function Likes({ user, setUser }) {

    const [painting, setPainting] = useState([]);
    const [refreshKey, setRefreshKey] = useState(0);
    const [isLogged] = useState(true);



    //get Painting info//
    useEffect(() => {

        setPainting([])

        for (let i = 0; i < user.paintingLiked.length; i++)
            axios.get(URLDayl + `paintingEntity/${user.paintingLiked[i]}`).then((response) => {

                let boolArray = response.data;
                boolArray.bool = false;
                setPainting(painting => [...painting, boolArray]);

            });




    }, [refreshKey]);



    console.log(painting)

    //Delete user like//
    const deleteLike = (paintId) => {


        axios.post(URLDayl + "users/updateFavoritePainting", { userId: user.id, favoriteId: paintId, operation: "remove" },
            { headers: { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json", "USER_PAINT": 0 } }
        )
            .then((response) => {


                console.log("remove")
                //update user
                axios.get(URLDayl + `users/${user.id}`, { headers: { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json" } }
                )
                    .then((response2) => {

                        setUser(response2.data)
                        setRefreshKey(oldKey => oldKey + 1);
                    });

            });




    }


    //afficher modale suppression//
    const handleBoolListTrue = (index) => {

        const tempBool = [...painting]

        tempBool[index].bool = true;
        setPainting(tempBool);
    }

    //cacher modale suppression//
    const handleBoolListFalse = (index) => {

        const tempBool = [...painting]

        tempBool[index].bool = false;
        setPainting(tempBool);

    }

    return (
        <div className="Likes">



            <div className="accountArtists">

                {painting.length > 0 &&
                    painting.map((paint, index) => {
                        return (
                            <div className="sliderOne owl-loaded ">
                                <div className="item" style={{ backgroundImage: `url(${URLDayl}images/content/mini/${paint.artist_idtf_intern}/${paint.generatedId})` }}>

                                    <div style={{ top: "193px", right: "150px" }}><ListModal user={user} isLogged={isLogged} listPaint={paint} /></div>
                                    <div className="bt-wrapper">

                                        {/* <div className="img-wrapper"> */}

                                        <div className="titre">
                                            <h3 style={{ fontStyle: "oblique", fontWeight: "600", fontSize: "14px", color: "white", marginBottom: "15px"
                                         }}>{paint.artist_name_tot}</h3>
                                            {paint.aw_title_Origin}
                                            <div className="date">{paint.aw_year}</div>
                                        </div>
                                        <div className="container-action">
                                            <Link className="bt-action voir" to={`../../view/OpusPage.js/${paint.id}`} state={paint.id}>

                                                voir

                                            </Link>
                                            <Link className="bt-action match" to={`../../view/MatchingPage.js/${paint.id}`} state={paint.id}>
                                                match
                                            </Link>
                                        </div>

                                    </div>
                                    <div className="name"></div>

                                    <input type="button" className="unfollow" onClick={() => handleBoolListTrue(index)} />

                                    {paint.bool === true &&
                                        <div className="deleteConfirm">
                                            <span className="deleteText">Supprimer?</span>
                                            <button type="button" className="yes closeTwoParentsNow" onClick={() => deleteLike(paint.id)} value="">Oui</button>
                                            <button type="button" className="no closeParentNow" onClick={() => handleBoolListFalse(index)} value="">Non</button>
                                        </div>}


                                </div>

                            </div>
                        )
                    })}

            </div>

        </div>

    );

}
