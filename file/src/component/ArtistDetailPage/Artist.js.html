    import React from "react";
import { useEffect, useState } from "react";
import axios from "axios";
import { URLDayl } from "../../services/constantes"
import { Link } from "react-router-dom";
// import verif from "../../webapp/img/icons/badge_verified-account_1.svg"
import { toast } from "react-toastify";
import { Tab, Tabs, TabList, TabPanel } from "react-tabs";
import ListPaintArtist from "./ListPaintArtist";
import SimilarArtist from "./SimilarArtist";
import Biography from "./Biography";


/**
 * Display information about the artist and child components in tabs
 * @emits - <strong>artist/byIdtf/{idtf}</strong> : get painter informations 
 * @emits - artistLiked/add(remove)FavoriteArtist/ : follow/unfollow artist
 * @emits - users/{id} : refresh user info
 */


export default function Artist({ user, setUser, paintId, isLogged }) {

	const [artist, setArtist] = useState([]);
	const [bg, setBg] = useState("");
	const [refreshKey, setRefreshKey] = useState(0);
	const [boolLike, setBoolLike] = useState(false);
	const [isActive, setIsActive] = useState(1);


	//get info painter//
	useEffect(() => {
		axios.get(URLDayl + `artist/byIdtf/${paintId}`, {}, { headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*" } })
			.then((response) => {
				setArtist(response.data);


			});


	}, [paintId]);



	//get random image for background//
	useEffect(() => {
		const randInt = Math.floor(Math.random() * (artist.numberOfPAinting - 1) + 1);


		if (randInt !== NaN) {
			setBg("image_" + randInt + ".jpg");

		}

	}, [artist]);

	//Changement onglet
	const toggleClass = (val) => {
		setIsActive(val);


	};

	//Test Follow//
	useEffect(() => {

		if (user && artist) {
			for (let i = 0; i < user.artistLiked.length; i++) {

				if (user.artistLiked[i] === artist.id) {
					setBoolLike(true);
				}

			}
		}
	}, [artist]);

	//Follow/Unfollow l'artiste//
	function follow() {
		if (user && artist) {


			if (boolLike === false)
				setBoolLike(true)
			else
				setBoolLike(false)




			if (user) {
				if (boolLike === false)
					axios.get(URLDayl + `artistLiked/addFavoriteArtist/${user.id}/${artist.id}`, {}, { headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*" } })
						.then((response) => {
							setBoolLike(true)


						});
				else
					axios.get(URLDayl + `artistLiked/deleteFavoriteArtist/${user.id}/${artist.id}`, {}, { headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*" } })
						.then((response) => {
							setBoolLike(false)

							console.log(response.data)
						});
			}
		}
		else {
			toast.warning("Connectez-vous pour suivre cet artiste!", {
				position: "top-right",
				autoClose: 3000,
				hideProgressBar: false,
				closeOnClick: true,
				pauseOnHover: true,
				draggable: true,
				progress: undefined,
			});
		}

		//refresh user
		if (user)
			axios.get(URLDayl + `users/${user.id}`, { headers: { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json" } }
			)
				.then((response2) => {

					setUser(response2.data)
				});

		setRefreshKey(oldKey => oldKey + 1);
	}


	return (
		<div className="Artist">

			{/* <!-- ******************************************************
			SECTION PRÉSENTATION DE L'ARTISTE - début
			****************************************************** --> */}
			<section className="center-col">

				<div className="cont-f-w artistBg" style={{ backgroundImage: `url(${URLDayl}images/content/${artist && artist.functionalId}/${bg})`, backgroundRepeat: "no-repeat", backgroundPosition: "center center", backgroundSize: "cover" }}></div>

				<div className="cont-n-w center-col artistBox">

					<ol className="topBcCenter mb-b" style={{ padding: "0 0 20px 0" }}>
						<li><Link to="../../">Accueil</Link></li>
						<li className="#"><a>Artistes</a></li>
						<li className="#">{artist.firstName}</li>
					</ol>
					{/* <!-- Suivre cet artiste - début --> */}
					<ul className="inlineList mb-c">
						{boolLike === false && <li><button className="followArtist" onClick={follow} type="button" value="#">&#x2b;&nbsp;&nbsp; Suivre</button></li>}
						{boolLike === true && <li><button className="followedArtist" onClick={follow} type="button" value="#">&#x2713;&nbsp;&nbsp; Suivi</button></li>}
					</ul>
					{/* <!-- Suivre cet artiste - fin -->
					<!-- Nom de l'artiste - début --> */}
					<h1>
						<span>{artist && artist.name}</span>
					</h1>
					{/* <!-- Nom de l'artiste - fin -->
					<!-- Données rapides de l'artiste - début --> */}
					<div className="artistQuickFacts">
						<span>{artist && artist.yearBirth} - {artist && artist.yearDeath}</span>
						<span>&nbsp;|&nbsp;</span>
						<span>{artist && artist.country}</span>
					</div>
					{/* <!-- Données rapides de l'artiste - fin -->
					<!-- Bio de l'artiste - début --> */}
					<p>{artist && artist.description}</p>
					{/* <!-- Bio de l'artiste - fin -->
					
					<!-- Suivre cet artiste - début --> */}


					<ul className="inlineList mt-b mb-a">
						<li><a onClick={(e) => setIsActive(1)} className={isActive === 1 ?"tabLink active" : "tabLink"}>Oeuvres</a></li>
						<li><a onClick={(e) => setIsActive(2)} className={isActive === 2 ?"tabLink active" : "tabLink"}>Biographie</a></li>
						<li><a onClick={(e) => setIsActive(3)} className={isActive === 3 ?"tabLink active" : "tabLink"}>Avis d'experts</a></li>
						<li><a onClick={(e) => setIsActive(4)} className={isActive === 4 ?"tabLink active" : "tabLink"}>Analyse chromatiques</a></li>
						<li><a onClick={(e) => setIsActive(5)} className={isActive === 5 ?"tabLink active" : "tabLink"}>Actualités</a></li>
					</ul>

					{/* <!-- Suivre cet artiste - fin --> */}
				</div>
			</section>
			{isActive === 1 && <div>
				<ListPaintArtist paintId={paintId} user={user} setUser={setUser} isLogged={isLogged} />
				<SimilarArtist paintId={paintId} />
			</div>
			}
			{isActive === 2 && <div>
				<Biography bg={bg} artist={artist}/>
			</div>
			}
			{isActive === 3 && <div>
			
			</div>
			}
			{isActive === 4 && <div>
			
			</div>
			}

			{/* <!-- SECTION PRÉSENTATION DE L'ARTISTE - fin --> */}

		</div>

	);
}
