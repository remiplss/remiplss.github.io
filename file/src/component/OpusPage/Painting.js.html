    import React from "react";
import { useRef, useEffect, useState, Suspense } from "react";

import axios from "axios";
import { URLDayl } from "../../services/constantes"
import PrismaZoom from 'react-prismazoom'
import { Link } from "react-router-dom";
import { accordion } from "../../webapp/js/js_source/accordion"
import ListModal2 from "../general/ListModal2";
import { EmailShareButton, FacebookShareButton, PinterestShareButton, TwitterShareButton, } from "react-share";
import { FacebookIcon, TwitterIcon, PinterestIcon, EmailIcon } from "react-share";
import { toast } from "react-toastify"
import ReactLoading from 'react-loading';

/**
 * This component display informations about a specific painting
 * @param {object} user - user object from api
 * @param {object} setUser - user setter
 * @param {boolean} isLogged - test if the user is logged
 * @param {number} paintId - selected paint Id
 * @emits - paintingEntity/${paintId} : get painting informations
 * @emits - paintingEntity/numberFavorite/${paintId} : get number of likes
 * @emits - api/painting_comment/save : post new comment
 * @emits - users/updateFavoritePainting : like/unlike painting
 */

export default function Painting({ user, setUser, paintId, isLogged }) {


	const [isLoading, setIsLoading] = useState(false);
	const [Painting, setPainting] = useState([]);
	const [zoomBool, setZoomBool] = useState(false);
	const [refreshKey, setRefreshKey] = useState(0);
	const [boolFav, setBoolFav] = useState(false);
	const [commentText, setCommentText] = useState("");
	const [shareUrlFb, setShareUrlFb] = useState(window.location.href);
	const [showKeyword, setShowKeyword] = useState(false)
	const [nLike, setNLike] = useState(0)
	const [boolAccordion, setBoolAccordion] = useState(false);


	accordion();

	function handleAccordion() {

		if (boolAccordion === true) {
			setBoolAccordion(false)
		}
		else {
			setBoolAccordion(true);
		}
	}

	//get Painting info//
	useEffect(() => {

		axios.get(URLDayl + `paintingEntity/${paintId}`).then((response) => {
			setIsLoading(true);
			setPainting(response.data);
			console.log(response.data)
			const tempResponse = response.data;
			//retrive keywords from string to array

			tempResponse.boolList = false;
			tempResponse.aw_keyw_ai = tempResponse.aw_keyw_ai.slice(1, -1)
			tempResponse.aw_keyw_ai = tempResponse.aw_keyw_ai.split(",")
			setPainting(tempResponse)
			setShowKeyword(true)
			//Get number like
			axios.get(URLDayl + `paintingEntity/numberFavorite/${paintId}`).then((response2) => {
				setNLike(response2.data);

			});

		});
		//Test painting liké par l'utilisateur
		if (user)
			user.paintingLiked.map((like) => {
				paintId === like && setBoolFav(true)
			})

	}, [paintId]);



	////////Add fav////////
	function handleClick() {
		const tempUser = user;
		if (isLogged === true) {
			if (boolFav === true) {
				setBoolFav(false);
				//delete a favorite painting
				axios.post(URLDayl + "users/updateFavoritePainting", { userId: user.id, favoriteId: paintId, operation: "remove" },
					{ headers: { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json", "USER_PAINT": 0 } }
				)
					.then((response) => {

						const removeIndex = tempUser.paintingLiked.indexOf(paintId);
						tempUser.paintingLiked.splice(removeIndex, 1);
						setUser(tempUser);
						updateLike()
						toast.success("Supprimée des favoris!", {
							position: "top-right",
							autoClose: 3000,
							hideProgressBar: false,
							closeOnClick: true,
							pauseOnHover: true,
							draggable: true,
							progress: undefined,
						});
					});

			}
			else {
				setBoolFav(true);
				//add a favorite painting
				axios.post(URLDayl + "users/updateFavoritePainting", { userId: user.id, favoriteId: paintId, operation: "add" },
					{ headers: { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json", "USER_PAINT": 0 } }
				)
					.then((response) => {



						tempUser.paintingLiked.push(paintId)
						setUser(tempUser)
						updateLike()
						toast.success("Ajoutée aux favoris!", {
							position: "top-right",
							autoClose: 3000,
							hideProgressBar: false,
							closeOnClick: true,
							pauseOnHover: true,
							draggable: true,
							progress: undefined,
						});

					});
			}



		}

	}
	function updateLike() {
		//update Nombres de like//
		axios.get(URLDayl + `paintingEntity/numberFavorite/${paintId}`).then((response) => {
			setNLike(response.data);
			console.log(response.data)

		});
	}

	useEffect(() => {
		const loadFile = async () => {
			// const { modal } = await import("../../webapp/js/modalBox.min.js");

			// modal();
		};
		loadFile();
	}, []);

	//afficher image zoom//
	const handleZoomTrue = () => {
		setZoomBool(false)
	}
	const handleZoomFalse = () => {
		setZoomBool(true)
	}


	//post comment//
	const handleSubmit = (e) => {

		e.preventDefault();
		if (isLogged === true) {
			axios.post(URLDayl + "api/painting_comment/save",
				{ userId: user.id, paintingId: paintId, content: commentText, statusComment: "ACTIVE", visibility: "PUBLIC" },
				{ headers: { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json" } }
			)
				.then((response) => {
					setRefreshKey(oldKey => oldKey + 1);

					toast.success("Commentaire posté!", {
						position: "top-right",
						autoClose: 3000,
						hideProgressBar: false,
						closeOnClick: true,
						pauseOnHover: true,
						draggable: true,
						progress: undefined,
					});
				});
		}
		else {
			toast.warn("Connectez vous pour poster un commentaire!", {
				position: "top-right",
				autoClose: 3000,
				hideProgressBar: false,
				closeOnClick: true,
				pauseOnHover: true,
				draggable: true,
				progress: undefined,
			});
		}


	}

					{/* <li className="#"><a>Oeuvres</a></li> */}
					<li className="#">{Painting.aw_title_Origin}</li>
				</ol>
				{/* <!-- Breadcrumbs - fin --> */}



				<div className="oeuvreTop">

					<div className="itemImage" style={{height: "400px"}}>

						{/* <!-- Lancer un matching avec cette oeuvre - début --> */}
						<form id="matchThisItem" className="matchSubAlt" method="post" action="">

							<Link to={`../../view/MatchingPage.js/${paintId}`} state={paintId} ><input type="submit" value="Match" /></Link>
						</form>
						{/* <!-- Lancer un matching avec cette oeuvre - fin --> */}

						{/* <!-- Lightbox image avec fonction agrandir/réduire - début --> */}

						<a className="zoomOpen" onClick={handleZoomFalse} data-modal="imgBox">
							<img style={{maxHeight: "400px"}} onLoad={e => setIsLoading(false)} className="show" src={URLDayl + `images/content/${Painting.artist_idtf_intern}/${Painting.generatedId}`} alt="Nom de l'oeuvre" />
							
						</a>
						{/*<!-- Trigger Modal. --> */}


						{zoomBool === true &&
							<div className="zoomBox" id="imgBox" style={{ overflowY: "hidden" }}>
								<button className="zoomCloseBtn" onClick={handleZoomTrue} type="button">x</button>

								<div className="modal-sandbox"></div>
								<PrismaZoom allowPan={true} allowZoom={true}>

									<img id="resizeMe" style={{ height: "auto ", maxWidth: "70% " }} src={URLDayl + `images/content/${Painting.artist_idtf_intern}/${Painting.generatedId}`} draggable="true" alt="Nom de l'oeuvre" />
								</PrismaZoom>
							</div>

						}

						{/* <!-- Lightbox image avec fonction agrandir/réduire - fin --> */}

					</div>

					<div className="itemDetails">
						<h1>{Painting.aw_title_Origin}, {Painting.aw_year}</h1>
						<h2><Link to={`../../view/ArtistDetailPage.js/${Painting.artist_idtf_intern}`} state={Painting.artist_idtf_intern} href="artiste.html">{Painting.artist_name_tot}</Link></h2>

						{/* <!-- LIEN PLUS D'INFOS SUR CETTE OEUVRE (WIKIART...) - début --> */}
						<a className="itemExtData" href={Painting.aw_link_ext} target="_blank">Plus d’informations sur {Painting.aw_desc_basic_source_1}</a>
						{/* <!-- LIEN PLUS D'INFOS SUR CETTE OEUVRE (WIKIART...) - fin --> */}

						{/* <!-- ENREGISTRER & PARTAGER LE MATCHING - début --> */}
						<ul className="hList">

							{/* <!-- BOUTONS LIKES - début --> */}
							<li>

								<input type="submit" id="submit" className={boolFav ? "itemLikeItYes" : "itemLikeIt"} onClick={() => handleClick()} value={nLike} />{/*<!-- Le membre n'aime pas déjà cette oeuvre --> */}

							</li>
							{/* <!-- BOUTONS LIKES - début --> */}

								<ListModal2 user={user} isLogged={isLogged} listPaint={Painting} />

									<TwitterIcon size={32} round style={{ marginRight: "1px" }} />

								<PinterestShareButton url={shareUrlFb} description={"Regarde l'oeuvre que j'ai trouvé sur Dayl.art !"} className="Demo__some-network__share-button" >
									<PinterestIcon size={32} round style={{ marginRight: "1px" }} />
								</PinterestShareButton>

								{/* <div className="smart-modal">
									<a data-target="shareThis" className="smart-modal-open-2"><span className="matchShare">Partager</span></a>
